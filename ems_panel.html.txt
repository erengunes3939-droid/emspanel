<script>
    let user = null; let tInt; let sTime;

    function toggleAuth(l) { document.getElementById('login-form').classList.toggle('hidden', !l); document.getElementById('register-form').classList.toggle('hidden', l); }

    function register() {
        let u = document.getElementById('reg-user').value; let p = document.getElementById('reg-pass').value;
        if(!u || !p) return alert("Hata!");
        localStorage.setItem('ems_'+u, JSON.stringify({u, p, c:0, h:0}));
        alert("Sisteme eklendin!"); toggleAuth(true);
    }

    function login() {
        let u = document.getElementById('user-in').value; let p = document.getElementById('pass-in').value;
        let data = JSON.parse(localStorage.getItem('ems_'+u));
        if(data && data.p === p) { user = data; start(); } else alert("Erişim Reddedildi!");
    }

    function start() {
        document.getElementById('auth-screen').classList.add('hidden');
        document.getElementById('main-app').style.display = 'grid';
        update();
    }

    function update() {
        document.getElementById('p-name').innerText = user.u;
        document.getElementById('p-coins').innerText = user.c;
        document.getElementById('total-h').innerText = user.h;
        localStorage.setItem('ems_'+user.u, JSON.stringify(user));
    }

    function showPage(p) {
        document.getElementById('page-dash').classList.toggle('hidden', p !== 'dash');
        document.getElementById('page-market').classList.toggle('hidden', p !== 'market');
        document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
    }

    function toggleDuty() {
        let btn = document.getElementById('duty-btn');
        if(!tInt) {
            sTime = new Date();
            tInt = setInterval(() => {
                let d = new Date() - sTime;
                let h = Math.floor(d/3600000); let m = Math.floor((d%3600000)/60000); let s = Math.floor((d%60000)/1000);
                document.getElementById('timer').innerText = `${h.toString().padStart(2,'0')}:${m.toString().padStart(2,'0')}:${s.toString().padStart(2,'0')}`;
            }, 1000);
            btn.innerText = "MESAİYİ DURDUR"; btn.style.background = "#555";
        } else {
            let hrs = Math.floor((new Date() - sTime) / 3600000);
            clearInterval(tInt); tInt = null;
            if(hrs > 0) { user.c += hrs; user.h += hrs; alert(hrs + " Coin kazandın!"); }
            document.getElementById('timer').innerText = "00:00:00";
            btn.innerText = "MESAİ BAŞLAT"; btn.style.background = "var(--neon-red)";
            update();
        }
    }

    function buy(p) { if(user.c >= p) { user.c -= p; alert("Başarılı!"); update(); } else alert("Coin yetersiz!"); }
    function logout() { location.reload(); }
</script>
